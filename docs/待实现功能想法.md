#### 数据库设计（3NF）

```sql
-- 信号配置表
CREATE TABLE signal_configs (
    id INT PRIMARY KEY AUTO_INCREMENT,
    hot_list_top INT,
    rank_jump_min INT,
    steady_rise_days INT,
    price_surge_min DECIMAL(5,2),
    volume_surge_min DECIMAL(5,2),
    volatility_surge_min DECIMAL(5,2),
    created_at TIMESTAMP,
    is_default BOOLEAN DEFAULT FALSE,
    config_name VARCHAR(100),
    description TEXT
);

-- 股票每日信号表
CREATE TABLE stock_signals (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    stock_code VARCHAR(10) NOT NULL,
    trade_date DATE NOT NULL,
    config_id INT NOT NULL,
    signal_type VARCHAR(50) NOT NULL,  -- 'hot_list', 'rank_jump', 'price_surge'等
    signal_label VARCHAR(100) NOT NULL,  -- '热点榜TOP100', '跳变↑3424'
    signal_score DECIMAL(5,2),
    signal_strength DECIMAL(5,4),
    -- 触发信号时的市场数据快照
    rank_value INT,
    price_change DECIMAL(10,2),
    turnover_rate DECIMAL(10,2),
    -- 未来N天的收益（用于效果评估，定时任务更新）
    return_1d DECIMAL(10,2),  -- 1天后收益
    return_3d DECIMAL(10,2),  -- 3天后收益
    return_5d DECIMAL(10,2),  -- 5天后收益
    return_7d DECIMAL(10,2),  -- 7天后收益
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_stock_date (stock_code, trade_date),
    INDEX idx_config (config_id),
    INDEX idx_signal_type (signal_type),
    INDEX idx_date (trade_date),
    FOREIGN KEY (config_id) REFERENCES signal_configs(id)
);

-- 信号组合记录表（一只股票可能同时有多个信号）
CREATE TABLE stock_signal_combinations (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    stock_code VARCHAR(10) NOT NULL,
    trade_date DATE NOT NULL,
    config_id INT NOT NULL,
    signal_count INT,  -- 信号数量
    signal_types JSON,  -- ['hot_list', 'rank_jump', 'price_surge']
    combined_score DECIMAL(5,2),
    combined_strength DECIMAL(5,4),
    return_1d DECIMAL(10,2),
    return_3d DECIMAL(10,2),
    return_5d DECIMAL(10,2),
    return_7d DECIMAL(10,2),
    created_at TIMESTAMP,
    INDEX idx_stock_date (stock_code, trade_date),
    INDEX idx_signal_count (signal_count),
    FOREIGN KEY (config_id) REFERENCES signal_configs(id)
);

-- 信号效果统计表（每日汇总）
CREATE TABLE signal_statistics (
    id INT PRIMARY KEY AUTO_INCREMENT,
    trade_date DATE NOT NULL,
    config_id INT NOT NULL,
    signal_type VARCHAR(50),
    trigger_count INT,  -- 触发次数
    avg_return_1d DECIMAL(10,2),  -- 平均1日收益
    avg_return_3d DECIMAL(10,2),
    avg_return_5d DECIMAL(10,2),
    avg_return_7d DECIMAL(10,2),
    win_rate_1d DECIMAL(5,2),  -- 1日胜率（涨的比例）
    win_rate_3d DECIMAL(5,2),
    win_rate_5d DECIMAL(5,2),
    win_rate_7d DECIMAL(5,2),
    max_return DECIMAL(10,2),  -- 最大收益
    min_return DECIMAL(10,2),  -- 最大亏损
    updated_at TIMESTAMP,
    UNIQUE KEY uk_date_config_type (trade_date, config_id, signal_type)
);
```
